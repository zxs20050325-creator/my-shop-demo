--- START OF FILE index.js --- // server.js - 完整后端 (集成Supabase数据库与统计优化) const express = require('express'); const cors = require('cors'); const path = require('path'); require('dotenv').config(); const app = express(); app.use(express.json()); app.use(cors()); // 静态资源 app.use('/images', express.static(path.join(__dirname, 'public', 'images'))); app.use('/frontend', express.static(path.join(__dirname, '..', 'frontend'))); // 数据库服务 (Supabase 封装在 db.js 中) const db = require('./db'); // 商品数据 (静态) const MOCK_PRODUCTS = [ { id: 1, name: "广惠寺华塔·微缩模型", price: 198, img: "/images/001.jpg", category: "数字藏品" }, { id: 2, name: "赵州桥·石刻镇纸", price: 88, img: "/images/002.jpg", category: "文创周边" }, { id: 3, name: "避暑山庄·全景画卷", price: 328, img: "/images/003.jpg", category: "数字画作" }, { id: 4, name: "正定古城·隐藏款", price: 999, img: "/images/004.jpg", category: "典藏精品" }, { id: 5, name: "蔚县剪纸·窗花", price: 58, img: "/images/005.jpg", category: "非遗手作" }, { id: 6, name: "唐山皮影·礼盒", price: 168, img: "/images/006.jpg", category: "非遗手作" } ]; // ------------------------------------------ // 1. 核心业务接口 (注册/登录/商品/行为) // ------------------------------------------ app.get('/api/products', (req, res) => { res.json({ items: MOCK_PRODUCTS }); }); app.get('/api/products/:id', (req, res) => { const p = MOCK_PRODUCTS.find(i => i.id == req.params.id); p ? res.json(p) : res.status(404).json({error: 'Not found'}); }); // 用户注册 app.post('/api/register', async (req, res) => { const { username, password } = req.body; try { const exists = await db.getUserByUsername(username); if(exists) return res.status(409).json({success:false, message:'用户已存在'}); await db.createUser(username, password); res.json({success:true}); } catch(e) { res.status(500).json({success:false}); } }); // 用户登录 app.post('/api/login', async (req, res) => { const { username, password } = req.body; try { const user = await db.getUserByUsername(username); if(user && user.password === password) res.json({success:true}); else res.status(401).json({success:false}); } catch(e) { res.status(500).json({success:false}); } }); // 行为跟踪 (写入日志表) app.post('/api/track', async (req, res) => { const { username, action, product } = req.body; try { await db.addLog(username || '游客', action, product || ''); res.json({success:true}); } catch(e) { res.status(500).json({success:false}); } }); // 购物车添加 app.post('/api/cart/add', async (req, res) => { const { username, product } = req.body; try { await db.addToCart(username, product); res.json({success:true}); } catch(e) { res.status(500).json({success:false}); } }); // ------------------------------------------ // 2. 管理员数据统计接口 (Admin Stats) - 深度优化版 // ------------------------------------------ app.get('/api/admin/stats', async (req, res) => { try { // A. 获取基础数据 const stats = await db.getLatestStats(); // 获取总营收/订单数 const logs = await db.getRecentLogs(2000); // 获取最近2000条日志进行分析 // B. 生成日期范围 (今年1月1日至今) const generateDateRange = () => { const now = new Date(); const start = new Date(now.getFullYear(), 0, 1); const end = new Date(now.getFullYear(), now.getMonth(), now.getDate()); const dates = []; let curr = new Date(start); while(curr <= end) { dates.push(new Date(curr)); curr.setDate(curr.getDate() + 1); } return dates; }; const dateRange = generateDateRange(); // C. 数据聚合逻辑 (按天分组) const logsByDate = {}; logs.forEach(log => { if(log.created_at) { // 截取 YYYY-MM-DD const dateStr = new Date(log.created_at).toISOString().split('T')[0]; if(!logsByDate[dateStr]) logsByDate[dateStr] = { count: 0, users: new Set() }; logsByDate[dateStr].count++; // 流量 +1 logsByDate[dateStr].users.add(log.username); // 记录用户 (去重) } }); // D. 填充图表数据 const dateLabels = []; const dailyTraffic = []; const dailyActiveUsers = []; dateRange.forEach(date => { const dateStr = date.toISOString().split('T')[0]; const label = `${date.getMonth()+1}-${date.getDate()}`; // 生成 M-D 格式 dateLabels.push(label); const dayData = logsByDate[dateStr] || { count: 0, users: new Set() }; dailyTraffic.push(dayData.count); dailyActiveUsers.push(dayData.users.size); }); // E. 计算其他统计 // 行为分布 let actionDist = [0, 0, 0, 0]; // 浏览, 加购, 支付, 其他 logs.forEach(l => { if(l.action.includes('浏览')) actionDist[0]++; else if(l.action.includes('加入')) actionDist[1]++; else if(l.action.includes('支付') || l.action.includes('结算')) actionDist[2]++; else actionDist[3]++; }); // 热门商品 (简单统计) const prodCount = {}; logs.filter(l => l.product).forEach(l => { prodCount[l.product] = (prodCount[l.product] || 0) + 1; }); const topProducts = Object.entries(prodCount) .sort((a,b) => b[1] - a[1]) .slice(0, 5); // F. 返回最终JSON res.json({ kpi: { revenue: stats.total_revenue || 0, orders: stats.total_orders || 0, visits: logs.length, activeUsers: new Set(logs.map(l => l.username)).size }, charts: { dateLabels, // [ "1-1", "1-2", ... ] dailyTraffic, // [ 10, 25, ... ] dailyActiveUsers, // [ 3, 5, ... ] topProducts, actionDistribution: actionDist }, logs: logs.slice(0, 50).map(l => ({ // 只返回最新50条用于列表展示 time: l.created_at, username: l.username, action: l.action, product: l.product })) }); } catch(err) { console.error("Admin stats error:", err); res.status(500).json({ error: "Stats failed" }); } }); // 管理员获取所有用户详细数据 app.get('/api/admin/users-data', async (req, res) => { try { const users = await db.getAllUsers(); const carts = await db.getAllCarts(); const favs = await db.getAllFavorites(); const logs = await db.getRecentLogs(200); res.json({ users, carts, favorites: favs, logs }); } catch(e) { res.status(500).json({error:e.message}); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
